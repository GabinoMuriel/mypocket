-- ==========================================
-- ENUMS & TYPES

CREATE TYPE public.transaction_type AS ENUM ('income', 'expense');

-- ==========================================
-- TABLES
-- ==========================================

-- ROLES TABLE (RBAC System)
CREATE TABLE public.roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL UNIQUE
);

-- Insert default roles
INSERT INTO public.roles (name) VALUES ('user'), ('admin');

-- PROFILES TABLE (Extended User Data)
CREATE TABLE public.profiles (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    first_name TEXT,
    last_name TEXT,
    phone TEXT,
    address TEXT,
    city TEXT,
    postal_code TEXT,
    birthdate DATE,
    avatar_url TEXT,
    role_id UUID REFERENCES public.roles(id) ON DELETE SET NULL,
    created_at TIMESTAMPTZ DEFAULT now()
);

-- CATEGORIES TABLE (Synced from DB to Store)
CREATE TABLE public.categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL,
    type public.transaction_type NOT NULL,
    is_system BOOLEAN DEFAULT false,
    icon TEXT,
    color TEXT,
    is_active BOOLEAN DEFAULT true
);

-- TRANSACTIONS TABLE (Core movement records)
CREATE TABLE public.transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
    category_id UUID REFERENCES public.categories(id) ON DELETE SET NULL,
    amount NUMERIC(12, 2) NOT NULL,
    type public.transaction_type NOT NULL,
    description TEXT,
    date TIMESTAMPTZ DEFAULT now()
);


-- ==========================================
-- FUNCTIONS & TRIGGERS
-- ==========================================

-- Function: Automatically create a profile for new users
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.profiles (id, role_id)
  VALUES (
    new.id, 
    (SELECT id FROM public.roles WHERE name = 'user' LIMIT 1)
  );
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Trigger: Fire profile creation after auth.users insertion
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();

-- Function: Apply default system category if none is provided
CREATE OR REPLACE FUNCTION public.set_default_category() RETURNS trigger AS $$
BEGIN
  IF NEW.category_id IS NULL THEN
    NEW.category_id := (
      SELECT id FROM public.categories 
      WHERE is_system = true AND name = 'Other' AND type = NEW.transaction_type -- Updated to match the new column
      LIMIT 1
    );
  END IF;

  IF NEW.category_id IS NULL THEN
    RAISE EXCEPTION 'Default system category "Other" not found for type %', NEW.transaction_type;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger: Fire default category assignment before transaction insertion
CREATE TRIGGER ensure_default_category
  BEFORE INSERT ON public.transactions
  FOR EACH ROW EXECUTE PROCEDURE public.set_default_category();

-- 1. Create a secure helper function to check if the current user is an admin
-- SECURITY DEFINER allows it to read the roles table while bypassing RLS, preventing infinite loops
CREATE OR REPLACE FUNCTION public.is_admin() RETURNS boolean AS $$
DECLARE
  admin_status boolean;
BEGIN
  SELECT EXISTS (
    SELECT 1 FROM public.profiles p
    JOIN public.roles r ON p.role_id = r.id
    WHERE p.id = auth.uid() AND r.name = 'admin'
  ) INTO admin_status;
  
  RETURN admin_status;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER SET search_path = public;

-- Rule: Prevent deletion of essential system categories
CREATE RULE protect_system_categories AS
ON DELETE TO public.categories
WHERE OLD.is_system = true
DO INSTEAD NOTHING;

-- TRANSACTIONS
CREATE POLICY "Users and Admins can view transactions" ON public.transactions 
FOR SELECT USING (auth.uid() = user_id OR public.is_admin());
CREATE POLICY "Users can create their own transactions" ON public.transactions FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Users can update their own transactions" ON public.transactions FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "Users can delete their own transactions" ON public.transactions FOR DELETE USING (auth.uid() = user_id);

-- 1. Securely fetch profiles combined with their auth.users email
CREATE OR REPLACE FUNCTION public.admin_get_users_with_emails()
RETURNS TABLE(
  id uuid, 
  email varchar, 
  first_name text, 
  last_name text, 
  phone text, 
  created_at timestamptz, 
  role_name text
)
SECURITY DEFINER SET search_path = public AS $$
BEGIN
  -- Strict verification using the is_admin() helper we made earlier
  IF NOT public.is_admin() THEN
    RAISE EXCEPTION 'Access denied: Must be an admin';
  END IF;

  RETURN QUERY
  SELECT
    p.id,
    au.email::varchar,
    p.first_name,
    p.last_name,
    p.phone,
    p.created_at,
    r.name AS role_name
  FROM public.profiles p
  JOIN auth.users au ON p.id = au.id
  LEFT JOIN public.roles r ON p.role_id = r.id
  ORDER BY p.created_at DESC;
END;
$$ LANGUAGE plpgsql;

-- 2. Securely delete a user from the authentication system
CREATE OR REPLACE FUNCTION public.admin_delete_user(target_user_id uuid)
RETURNS void
SECURITY DEFINER SET search_path = public AS $$
BEGIN
  -- Strict verification
  IF NOT public.is_admin() THEN
    RAISE EXCEPTION 'Access denied: Must be an admin';
  END IF;

  -- Deleting from auth.users triggers the ON DELETE CASCADE on your public.profiles table
  DELETE FROM auth.users WHERE id = target_user_id;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION public.admin_update_user(
  target_user_id UUID,
  p_first_name TEXT,
  p_last_name TEXT,
  p_phone TEXT,
  p_address TEXT,
  p_city TEXT,
  p_postal_code TEXT,
  p_birthdate DATE
)
RETURNS boolean AS $$
BEGIN
  -- Verify the executor is an admin using your existing helper
  IF NOT public.is_admin() THEN
    RAISE EXCEPTION 'Access denied: Admin privileges required';
  END IF;

  -- Update the extended profile data
  UPDATE public.profiles
  SET first_name = p_first_name,
      last_name = p_last_name,
      phone = p_phone,
      address = p_address,
      city = p_city,
      postal_code = p_postal_code,
      birthdate = p_birthdate
  WHERE id = target_user_id;

  RETURN true;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
